```javascript
getTreeOptions() {
  const hasReview = this.mergeCandidates.length > 0 || this.duplicateSubtrees.length > 0;
  if (!hasReview) return {};
  return { reviewMode: true, mergeCandidates, duplicateSubtrees, onMerge, onKeep, onRemoveDupe, onKeepDupe };
},
rerenderTree() {
  this.$nextTick(() => {
    const container = this.$refs.treeContainer;
    if (!container) return;
    container.innerHTML = '';
    renderTree(this.cleanTree || this.tree, container, 0, this.getTreeOptions());
  });
},
```

    **Step 2 — Add onToggleReclassify to getTreeOptions():**

    Update `getTreeOptions()` to always include the reclassify callback (not gated by reviewMode), since the toggle should appear in the checked state regardless of merge candidates:

    ```javascript
    getTreeOptions() {
      const opts = {};
      const hasReview = this.mergeCandidates.length > 0 || this.duplicateSubtrees.length > 0;
      if (hasReview) {
        opts.reviewMode = true;
        opts.mergeCandidates = this.mergeCandidates;
        opts.duplicateSubtrees = this.duplicateSubtrees;
        opts.onMerge = (candidate) => this.approveMerge(candidate);
        opts.onKeep = (candidate) => this.keepSeparate(candidate);
        opts.onRemoveDupe = (entry) => this.removeDuplicate(entry);
        opts.onKeepDupe = (entry) => this.keepDuplicate(entry);
      }
      opts.onToggleReclassify = (folderName) => {
        const next = new Set(this.reclassifyFolders);
        if (next.has(folderName)) {
          next.delete(folderName);
        } else {
          next.add(folderName);
        }
        this.reclassifyFolders = next;  // Reassign for Alpine reactivity (Pitfall 3)
        this.rerenderTree();  // Re-render to update badge state
      };
      return opts;
    },
    ```
