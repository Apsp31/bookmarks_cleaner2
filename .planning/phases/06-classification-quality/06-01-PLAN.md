    **Step 3 — Add classifyByPath function (per D-01, D-02):**

    Add to src/classifier.js, after classifyByMetadata and before classifyNode:

    ```javascript
    export function classifyByPath(url) {
      let pathname, hostname;
      try {
        const parsed = new URL(url);
        pathname = parsed.pathname.toLowerCase();
        hostname = parsed.hostname.toLowerCase();
      } catch {
        return null;
      }

      // Subdomain hints (raw hostname, NOT www-stripped — per D-02, Pitfall 5)
      if (hostname.startsWith('docs.') || hostname.startsWith('documentation.')) return 'Reference';
      if (hostname.startsWith('blog.'))  return 'News';
      if (hostname.startsWith('shop.') || hostname.startsWith('store.')) return 'Shopping';

      // Path hints (per D-01)
      if (pathname.startsWith('/docs/') || pathname.startsWith('/docs') || pathname.startsWith('/documentation/')) return 'Reference';
      if (pathname.startsWith('/blog/') || pathname.startsWith('/blog') || pathname.startsWith('/news/') || pathname.startsWith('/news')) return 'News';
      if (pathname.startsWith('/shop/') || pathname.startsWith('/shop') || pathname.startsWith('/store/') || pathname.startsWith('/store')) return 'Shopping';
      if (pathname.startsWith('/api/') || pathname.startsWith('/api')) return 'Development';

      return null;
    }
    ```

    export function classifyNode(node) {
